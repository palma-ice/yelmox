# ISMIP6-2300 experiments

For a description of all experiments see here:
https://www.climate-cryosphere.org/wiki/index.php?title=ISMIP6-Projections2300-Antarctica#Initialization.2C_historical_run.2C_control_run.2C_and_projection_runs


# Spinup, 32KM:
```
res=32km
grid_name=ANT-32KM
#res=16km
#grid_name=ANT-16KM

runopt='-s -q short -w 24'
#runopt='-r'

fldr=tmp/ismip6/spinup_${res}_111

paropt="ydyn.ssa_lat_bc='floating' ytopo.kt=2.5e-3 ymat.enh_shear=2 yneff.p=0.5 ytill.cf_ref=1.0"

jobrun ./runylmox ${runopt} -e ismip6 -n par/yelmo_ismip6_Antarctica.nml -- -o ${fldr} -p ctrl.run_step="spinup" spinup.equil_method="opt" yelmo.grid_name=${grid_name} ${paropt}

```

# Spinup 16KM from an already optimized 32KM run:
```
res=16km
grid_name=ANT-16KM

runopt='-s -q short -w 24'
runopt='-r'

file_restart=/p/tmp/robinson/ismip6/spinup_32km_111/0/yelmo_restart.nc

fldr=tmp/ismip6/spinup_${res}_112

paropt="ydyn.ssa_lat_bc='floating' ytopo.kt=2.5e-3 ymat.enh_shear=2 yneff.p=0.5 ytill.cf_ref=1.0"

paropt2="yelmo.restart=${file_restart} opt.cf_time=2e3 opt.tf_time=2e3 opt.rel_time2=0.0 spinup.time_end=6e3"

jobrun ./runylmox ${runopt} -e ismip6 -n par/yelmo_ismip6_Antarctica.nml -- -o ${fldr} -p ctrl.run_step="spinup" spinup.equil_method="opt" yelmo.grid_name=${grid_name} yelmo.dt_min=0.1 ${paropt2} ${paropt}

```

# Include ISMIP6 formatting?

# Experiments
expname=ctrlAE
expname=expAE01
expname=expAE02
expname=expAE03
expname=expAE04
expname=expAE05
expname=expAE06
expname=expAE07
expname=expAE08
expname=expAE09
expname=expAE10
expname=expAE11
expname=expAE12
expname=expAE13

# Or run all experiments
